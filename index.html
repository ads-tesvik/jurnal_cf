  return DB.journalStaff.some(s=>s.userId===userId&&(!slug||s.journalSlug===slug)&&(!role||s.jRole===role));

function generatePDF(a,anon=false){

  const {jsPDF}=window.jspdf;const doc=new jsPDF();const j=getJ(a.journalSlug);

  let y=20;

  doc.setFillColor(15,30,61);doc.rect(0,0,210,17,'F');

  doc.text(anon?'[ANONİM HAKEM KOPYASı]':'[TAM KOPYA]',150,10);

  y=26;

  const tl=doc.splitTextToSize(a.title,170);doc.text(tl,20,y);y+=tl.length*7+4;

  doc.setTextColor(15,30,61);doc.setFontSize(11);doc.setFont('helvetica','bold');doc.text('Özet',20,y);y+=5;

  doc.setFont('helvetica','normal');doc.setFontSize(10);doc.setTextColor(50,50,70);

  const abl=doc.splitTextToSize(a.abstract||'',170);doc.text(abl,20,y);y+=abl.length*5+4;

  if(a.abstractEn){doc.setFont('helvetica','bold');doc.setFontSize(11);doc.setTextColor(15,30,61);doc.text('Abstract',20,y);y+=5;doc.setFont('helvetica','normal');doc.setFontSize(10);doc.setTextColor(50,50,70);const ael=doc.splitTextToSize(a.abstractEn,170);doc.text(ael,20,y);y+=ael.length*5+4;}

  if(a.keywords){doc.setFont('helvetica','bold');doc.setFontSize(10);doc.setTextColor(15,30,61);doc.text('Anahtar Kelimeler: ',20,y);doc.setFont('helvetica','normal');doc.setTextColor(50,50,70);doc.text(a.keywords,65,y);y+=8;}
